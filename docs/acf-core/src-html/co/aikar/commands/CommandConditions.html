024package co.aikar.commands;
025
026import co.aikar.commands.annotation.Conditions;
027import com.google.common.collect.HashBasedTable;
028import com.google.common.collect.Maps;
029import com.google.common.collect.Table;
030
031import java.util.Map;
032
033@SuppressWarnings("BooleanMethodIsAlwaysInverted") // No IDEA, you are wrong
034public class CommandConditions <
035        I extends CommandIssuer,
036        M extends CommandManager,
037        CC extends ConditionContext,
038        CEC extends CommandExecutionContext<CEC, I>,
039        PCC extends ParameterConditionContext<?, CEC, I>
040    > {
041    M manager;
042    Map<String, Condition<CC>> conditions = Maps.newHashMap();
043    Table<Class<?>, String, ParameterCondition<PCC>> paramConditions = HashBasedTable.create();
044    CommandConditions(M manager) {
045        this.manager = manager;
046    }
047
048    Condition<CC> addCondition(String id, Condition<CC> handler) {
049        return this.conditions.put(id.toLowerCase(), handler);
050    }
051
052    <P> ParameterCondition addCondition(Class<P> clazz, String id, ParameterCondition<PCC> handler) {
053        return this.paramConditions.put(clazz, id.toLowerCase(), handler);
054    }
055
056    boolean validateConditions(CommandOperationContext context, CEC cec) {
057        Conditions conditions = cec.getParam().getAnnotation(Conditions.class);
058        return conditions == null || validateConditions(conditions, context, cec);
059    }
060
061    boolean validateConditions(CommandOperationContext context) {
062        RegisteredCommand cmd = context.getRegisteredCommand();
063        Conditions conditions = cmd.method.getAnnotation(Conditions.class);
064        if (conditions != null) {
065            if (!validateConditions(conditions, context)) {
066                return false;
067            }
068        }
069        return validateConditions(cmd.scope, context);
070    }
071
072
073    private boolean validateConditions(BaseCommand scope, CommandOperationContext operationContext) {
074        Conditions conditions = scope.getClass().getAnnotation(Conditions.class);
075        //noinspection SimplifiableIfStatement
076        if (!validateConditions(conditions, operationContext)) {
077            return false;
078        }
079        return scope.parentCommand == null || validateConditions(scope.parentCommand, operationContext);
080    }
081
082    private boolean validateConditions(Conditions condAnno, CommandOperationContext context) {
083        if (condAnno == null) {
084            return true;
085        }
086        //noinspection unchecked
087        CC conditionContext = (CC) this.manager.createConditionContext(context, condAnno);
088        String conditions = this.manager.getCommandReplacements().replace(condAnno.value());
089        for (String cond : ACFPatterns.PIPE.split(conditions)) {
090            String[] split = ACFPatterns.EQUALS.split(cond, 2);
091            Condition<CC> condition = this.conditions.get(split[0].toLowerCase());
092
093            if (!condition.validateCondition(conditionContext)) {
094                return false;
095            }
096        }
097
098        return true;
099    }
100
101    private boolean validateConditions(Conditions condAnno, CommandOperationContext context, CEC execContext) {
102        if (condAnno == null) {
103            return true;
104        }
105        //noinspection unchecked
106        ParameterConditionContext conditionContext = this.manager.createConditionContext(context, execContext, condAnno);
107        String conditions = this.manager.getCommandReplacements().replace(condAnno.value());
108        for (String cond : ACFPatterns.PIPE.split(conditions)) {
109            String[] split = ACFPatterns.EQUALS.split(cond, 2);
110            ParameterCondition condition;
111            Class<?> cls = execContext.getParam().getClass();
112            String id = split[0].toLowerCase();
113            do {
114                condition = this.paramConditions.get(cls, id);
115                if (condition == null && cls.getSuperclass() != null && cls.getSuperclass() != Object.class) {
116                    cls = cls.getSuperclass();
117                } else {
118                    break;
119                }
120            } while (cls != null);
121
122            //noinspection unchecked
123            if (condition != null && !condition.validateCondition(conditionContext)) {
124                return false;
125            }
126        }
127
128        return true;
129    }
130
131    interface Condition <CC extends ConditionContext> {
132        boolean validateCondition(CC context);
133    }
134    interface ParameterCondition <PCC extends ParameterConditionContext> {
135        boolean validateCondition(PCC context);
136    }
137}
